export default function reducer(state = initialState, action) {
  switch (action.type) {
    case actionType.ADD_NUMBER:
      return {...state, counter: state.counter + action.num}
    case actionType.SUB_NUMBER:
      return {...state, counter: state.counter + action.num}
    case actionType.MUL_NUMBER:
      return {...state, counter: state.counter * action.num}
    default:
      return state;
  }
}
