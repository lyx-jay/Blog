  url += (url.indexOf('?') === -1 ? '?' : "&");
  let key:keyof Params;
  for (key in params) {
    if (Object.prototype.hasOwnProperty.call(params, key)) {
      url += (encodeURIComponent(key) + '=' + encodeURIComponent(params[key]) + '&')
    }

  url = url.replace(/&$/g, ''); // å»é™¤æ‹¼æ¥åurlä¸­çš„æœ«å°¾&
